-- query 7
-- Mostrare gli sponsor che hanno sponsorizzato performance in almeno 3 giorni diversi.

SELECT 
    s.nome AS nome_sponsor, 
    COUNT(DISTINCT p.data_esibizione) AS giorni_distinti
FROM 
    Sponsor s
JOIN 
    Sponsorizzazione sp ON s.id_sponsor = sp.id_sponsor
JOIN 
    Performance p ON sp.id_performance = p.id_performance
GROUP BY 
    s.id_sponsor, s.nome
HAVING 
    COUNT(DISTINCT p.data_esibizione) >= 3;

----------- query 9 -------------------------

Mostrare la variazione percentuale di incasso giorno per giorno.

SELECT 
    giorno,
    incasso_odierno,
    incasso_precedente,
    CASE 
        WHEN incasso_precedente IS NULL THEN 'N/A' -- Gestisce il primo giorno del festival
        ELSE ROUND(((incasso_odierno - incasso_precedente) / incasso_precedente) * 100, 2) 
    END AS variazione_percentuale
FROM (
    SELECT 
        DATE(pa.data_pagamento) AS giorno,
        SUM(pa.importo) AS incasso_odierno,
        LAG(SUM(pa.importo)) OVER (ORDER BY DATE(pa.data_pagamento)) AS incasso_precedente
    FROM 
        Pagamento pa
    WHERE 
        pa.stato = 'completato'
    GROUP BY 
        DATE(pa.data_pagamento)
) AS tabella_trend
ORDER BY 
    giorno;

JOIN Biglietto big ON perf.id_performance = big.id_performance
JOIN Pagamento pag ON big.id_biglietto = pag.id_biglietto
WHERE 
    pag.stato = 'completato'
GROUP BY 
    a.id_artista, a.nome
HAVING 
    -- 1. Almeno 3 performance
